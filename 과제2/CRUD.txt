create table room( room_id int(20) NOT NUll,room_floor int(20) NULL,
    room_size int(20) NULL,
    bedroom_number int(20) NULL,
    bathroom_number int(20) NULL,
    primary key (room_id)
);

create table property(
    property_id int(20) NOT NULL,
    time_slot_id int(20) NOT NULL,
    end_time int(20) NOT NULL,
    start_time int(20) NOT NULL,
    property_cost int(20) NULL,
    floor_plan_number int(20) NULL,
    school_district_number int(20) NULL,
    property_address int(20) NULL,
    primary key (property_id),
    foreign key (time_slot_id) references time_slot(time_slot_id)
);

create table time_slot(
    time_slot_id int(20) NOT NULL,
    start_time int(20) NOT NULL,
    end_time int(20) NOT NULL,
    primary key (time_slot_id)
);

create table seller(
    seller_id int(20) NOT NULL,
    seller_name varchar(20) NULL,
    seller_phone_number int(20) NULL,
    primary key (seller_id)
);

create table buyer(
    buyer_id int(20) NOT NULL,
    buyer_name varchar(20) NULL,
    buyer_phone_number int(20) NULL,
    primary key (buyer_id)
);

create table agent(
    agent_id int(20) NOT NULL,
    agent_name varchar(18) NULL,
    agent_phone_number int(20) NULL,
    primary key (agent_id)
);

create table photo(
    interior_photo int(20) NOT NULL,
    exterior_photo int(20) NOT NULL,
    photo_id int(20) NOT NULL,
    primary key (photo_id)
);

create table match(
    room_id int(20) NOT NULL,
    property_id int(20) NOT NULL,
    primary key (room_id, property_id),
    foreign key (room_id) references room(room_id),
    foreign key (property_id) references property(property_id)
);

create table consider(
    interior_photo int(20) NOT NULL,
    exterior_photo int(20) NOT NULL,
    photo_id int(20) NOT NULL,
    property_id int(20) NOT NULL,
    primary key (interior_photo, exterior_photo, photo_id, property_id),
    foreign key (interior_photo) references photo(interior_photo),
    foreign key (exterior_photo) references photo(exterior_photo),
    foreign key (photo_id) references photo(photo_id),
    foreign key (property_id) references property(property_id)
);

create table deal(
    property_id int(20) NOT NULL,
    agent_id int(20) NOT NULL,
    primary key (property_id, agent_id),
    foreign key (property_id) references property(property_id),
    foreign key (agent_id) references agent(agent_id)
);

create table sell(
    seller_id int(20) NOT NULL,
    agent_id int(20) NOT NULL,
    primary key (seller_id, agent_id),
    foreign key (seller_id) references seller(seller_id),
    foreign key (agent_id) references agent(agent_id)
);

create table buy(
    buyer_id int(20) NOT NULL,
    agent_id int(20) NOT NULL,
    primary key (buyer_id, agent_id),
    foreign key (buyer_id) references buyer(buyer_id),
    foreign key (agent_id) references agent(agent_id)
);
